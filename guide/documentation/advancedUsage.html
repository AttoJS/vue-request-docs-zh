# 高级用法

我们基于前面章节提到的基本用法做了进一步的封装，目前实现了 分页，LoadMore 这些场景。如果你有好的建议，欢迎到 github在新窗口打开 发起 ISSUE 或者 PR。

# 分页扩展 New in beta.8

我们会自动帮你管理分页信息，如：当前页数、每页条数、总条数等。你只需要使用我们提供的 changeCurrent() 和 changePageSize() 即可轻松管理分页的数据。

注意：我们会在 params 数组第一位注入分页所需的参数。

更多可用配置请查看 分页扩展 API

# 用法

import { usePagination } from 'vue-request';

# 例子

你也可以配合 ant-design-vue 使用。这里我们提供了一个 demo，点击查看在新窗口打开

点击查看代码
import { defineComponent } from 'vue';
import { usePagination } from 'vue-request';
import Mock from 'mockjs';
import { Loading, Pagination } from 'vant';
import 'vant/lib/pagination/style';
import 'vant/lib/loading/style';

const getData = (): string[] => Mock.mock({ 'array|5': ['@name'] }).array;
function testService() {
  return new Promise<{ total: number; result: string[] }>(resolve => {
    setTimeout(() => {
      resolve({
        total: 100,
        result: getData(),
      });
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data, current, total, loading, pageSize, changeCurrent } = usePagination(testService);

    return () => (
      <div>
        <div style="height: 140px">
          {loading.value ? (
            <Loading />
          ) : (
            <ul>
              {data.value?.result.map(i => (
                <li key={i}>{i}</li>
              ))}
            </ul>
          )}
        </div>
        <Pagination
          modelValue={current.value}
          total-items={total.value}
          show-page-size={pageSize.value}
          force-ellipses
          onChange={(page: number) => changeCurrent(page)}
        />
      </div>
    );
  },
});

# LoadMore 扩展 New in beta.11

注意

LoadMore 模式下，仅支持 请求函数 形式的 Service

我们会自动帮你管理列表数据。Service 返回的数据结构必须包含 {list: LR[]} ，如果不满足，可以通过 listKey 指定别的键名 或者 formatResult() 转换一次。接着通过我们返回的 dataList 变量，即可获取到请求的数据列表集合，然后使用 loadMore() 触发加载更多

此外，我们会在 Service 函数的第一个参数传入 {data: R, dataList: LR} 你可以根据这些参数决定下一次 Service的请求参数。

更多可用配置请查看 LoadMore 扩展 API

# 用法

import { useLoadMore } from 'vue-request';

# 例子

你也可以配合 ant-design-vue 使用。这里我们提供了一个 demo，点击查看在新窗口打开

点击查看代码
import { defineComponent, ref } from 'vue';
import { useLoadMore } from 'vue-request';
import Mock from 'mockjs';
import { Cell, List, Button, PullRefresh } from 'vant';
import 'vant/es/list/style';
import 'vant/es/cell/style';
import 'vant/es/button/style';
import 'vant/es/pull-refresh/style';

function generateData(current: number, total = 10) {
  let list: string[] = [];
  if (current <= total) {
    list = Mock.mock({
      'list|10': ['@name'],
    }).list;
  } else {
    list = [];
  }
  return {
    current,
    total,
    list,
  };
}
type APIReturnType = ReturnType<typeof generateData>;
type rawDataType = {
  data: APIReturnType;
  dataList: APIReturnType['list'];
};

function testService(rawData: rawDataType) {
  const current = (rawData?.data?.current || 0) + 1;
  return new Promise<APIReturnType>(resolve => {
    setTimeout(() => {
      resolve(generateData(current));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { loadMore, loadingMore, dataList, noMore, refreshing, reload, refresh } = useLoadMore<
      APIReturnType,
      any,
      APIReturnType['list']
    >(testService, {
      isNoMore: d => d?.current >= d?.total,
    });

    const onRefresh = () => {
      document.querySelector('.list')?.scrollTo({ top: 0, behavior: 'smooth' });
      refresh();
    };

    return () => (
      <div>
        <Button
          style="margin-bottom: 20px"
          type="warning"
          size="mini"
          onClick={() => {
            reload();
          }}
        >
          Reload
        </Button>
        <Button style="margin-bottom: 20px" type="warning" size="mini" onClick={onRefresh}>
          Refresh
        </Button>
        <div class="list" style="height: 300px; overflow-y: scroll">
          <PullRefresh modelValue={refreshing.value} onRefresh={() => refresh()}>
            <List
              loading={loadingMore.value}
              loading-text="loading..."
              finished={noMore.value}
              finished-text="no more"
              onLoad={() => {
                loadMore();
              }}
            >
              {dataList.value?.map((i, idx) => (
                <Cell key={i} value={`${idx}  ${i}`} />
              ))}
            </List>
          </PullRefresh>
        </div>
      </div>
    );
  },
});
