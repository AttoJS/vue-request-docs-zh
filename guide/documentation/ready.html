# 依赖请求

有些时候，你可能会遇到网络请求相互依赖的情况。例如 B 请求的请求参数，依赖 A 请求的返回结果。这时，你可以使用 ready 来处理这种依赖关系。

如下所示，ready 接收一个 Ref<boolean>，当结果为 true 时，第二个请求将会发出。

import { computed, defineComponent } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

function loginService() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve('token');
    }, 1000);
  });
}

function getUser() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(Mock.mock('@name'));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data: token, loading: tokenLoading, run } = useRequest(loginService, { manual: true });
    const { data, loading } = useRequest(getUser, {
      ready: computed(() => !!token.value),
    });
    return () => (
      <div>
        <div>
          {data.value
            ? `User Name: ${data.value}`
            : loading.value
            ? 'loading...'
            : 'Please login first'}
        </div>
        <br />
        <button disabled={!!token.value} onClick={() => run()}>
          {tokenLoading.value ? 'Login...' : 'Login'}
        </button>
      </div>
    );
  },
});
